d1 = int(input("введите длину 1"))
d2 = int(input("введите длину 2"))
h1 = int(input("введите ширину 1"))
h2 = int(input("введите ширину 2"))
y = d1 * h1
x = d2 * h2
if y > x:
    print("прямоугольник 1 больше")
if y < x:
    print("прямоугольник 2 больше")
if y == x:
    print("прямоугольники равны")
